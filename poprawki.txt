----------------------- KOD JS DLA SLIDERA ---------------------------------------

<script>
    const slider = document.querySelector('.slider');
    const images = document.querySelectorAll('.slider__image');
    const dots = document.querySelectorAll('.slider__dot');

    let current = 0;
    let timer;

    function startSlider() {
        timer = setInterval(() => {
            // Przejdź do następnego zdjęcia
            current = (current + 1) % images.length;
            updateSlider();
        }, 5000);
    }

    function updateSlider() {
        // Ukryj wszystkie zdjęcia
        images.forEach((image) => (image.style.display = 'none'));
        // Pokaż aktualne zdjęcie
        images[current].style.display = 'block';
        // Usuń aktywny stan z wszystkich kropek
        dots.forEach((dot) => dot.classList.remove('slider__dot--active'));
        // Ustaw aktywny stan na aktualnej kropce
        dots[current].classList.add('slider__dot--active');
    }

    // Obsługa kliknicę w kropki
    dots.forEach((dot, index) => {
        dot.addEventListener('click', () => {
            // Przerwij automatyczne przewijanie
            clearInterval(timer);
            // Ustaw aktualne zdjęcie na klikniętą kropkę
            current = index;
            // Aktualizuj slider
            updateSlider();
            // Rozpocznij automatyczne przewijanie od nowa
            startSlider();
        });
    });

    // Rozpocznij automatyczne przewijanie
    startSlider();
</script>
